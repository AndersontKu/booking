var data = sheet.getDataRange().getValues(); var config = {}; data.forEach(function(row) { if (row[0]) config[String(row[0]).trim()] = String(row[1]).trim(); }); return config; } // ─── 共用 JSON 回傳 ─────────────────────────────────────────── function jsonResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } // ─── 計算指定場次正取人數 ───────────────────────────────────── // 依傳入 courseDate 篩選，不同日期獨立計算 // → 修改 config 的 COURSE_DATE，人數自動對新場次重新計算 function getRegCount(courseDate) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('報名資料'); } function isValidEmail(email) { return /^[^\s@]+@[^\s@]+\.[^\s@]{2,}$/.test(String(email).trim()); } function isValidName(name) { var n = String(name).trim(); return n.length >= 1 && n.length <= 30; } // ─── 安全：Sanitize 輸入 ───────────────────────────────────── function sanitize(str) { return String(str || '').replace(/[<>"'`\\]/g, '').trim(); } // ─── 安全：Rate Limit（同 Email 60 秒限一次）───────────────── var RATE_LIMIT_SECONDS = 60; }); } // ─── GET：前端載入 ?action=checkFull ───────────────────────── // 修復：全函式包 try/catch，永遠回 JSON，不再回空頁（「載入中」的根本原因） function doGet(e) { try { var action = (e && e.parameter && e.parameter.action) ? e.parameter.action : ''; } var count = getRegCount(config.COURSE_DATE); var remaining = Math.max(0, maxSeats - count); return jsonResponse({ isFull: count >= maxSeats, courseName: config.COURSE_NAME || '課程名稱', courseDate: config.COURSE_DATE || '日期待定', coursePrice: config.COURSE_PRICE || '費用待定', maxSeats: maxSeats, fuseSeats: fuseSeats, remaining: remaining }); } return jsonResponse({ result: 'error', message: 'unknown_action' }); } catch(err) { Logger.log('doGet error: ' + err.toString()); return jsonResponse({ result: 'error', message: 'server_error' }); } } // ─── POST：提交報名 / 候補 ──────────────────────────────────── function doPost(e) { try { var config = getConfig(); } // 後端格式驗證 if (!isValidName(params.name)) return jsonResponse({ result: 'error', message: '姓名格式錯誤' }); if (!isValidPhone(params.phone)) return jsonResponse({ result: 'error', message: '電話格式錯誤' }); if (!isValidEmail(params.email)) return jsonResponse({ result: 'error', message: 'Email 格式錯誤' }); } // 併發鎖（防超賣） var lock = LockService.getScriptLock(); try { lock.waitLock(10000); } catch(le) { return jsonResponse({ result: 'error', message: '系統忙碌，請稍後再試' }); } try { var count = getRegCount(config.COURSE_DATE); var newCount = (status === '✅正取') ? count + 1 : count; var newIsFull = newCount >= maxSeats; // 熔斷通知管理員 if (newCount >= fuseSeats && config.ADMIN_EMAIL) { try { MailApp.sendEmail({ to: config.ADMIN_EMAIL, subject: '【御景家】⚠️ 報名額滿通知：' + config.COURSE_DATE, body: config.COURSE_DATE + ' 報名已達熔斷上限（' + fuseSeats + ' 人）。\n表單已自動關閉，請前往後台確認，並視需求開啟下一場次。\n\n御景家報名系統' });
